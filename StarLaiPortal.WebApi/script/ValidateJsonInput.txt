  CREATE OR ALTER PROCEDURE [dbo].[ValidateJsonInput]
  @type nvarchar(50),
  @json nvarchar(max)
  AS
  BEGIN

      IF (@type = 'PickListDetailsActual')
      BEGIN
          DECLARE @PickDocNum NVARCHAR(50);
          DECLARE @ItemCode   NVARCHAR(50);

          -- D2 : quantities from the JSON being submitted NOW (not from DB)    
		SELECT TOP 1 @PickDocNum = D3.DocNum, @ItemCode = D1.ItemCode
          FROM PickListDetails D1
          INNER JOIN (
              SELECT SOBaseId, SUM(CAST(PickQty AS DECIMAL(18,4))) AS PickQty
              FROM OPENJSON(@json, '$.PickListDetailsActuals')
              WITH (
                  SOBaseId INT           '$.SOBaseId',
                  PickQty  DECIMAL(18,4) '$.PickQty'
              )
              GROUP BY SOBaseId
          ) D2 ON D1.SOBaseId = D2.SOBaseId
              AND D1.PickList <> CAST(JSON_VALUE(@json, '$.PickOid') AS INT)
          INNER JOIN PickList D3          ON D1.PickList  = D3.OID
          INNER JOIN SalesOrderDetails D4 ON D2.SOBaseId  = D4.OID
          WHERE D3.GCRecord IS NULL
            AND D3.Status IN (1, 5)       -- Submitted, Posted
            AND D1.GCRecord IS NULL
            AND D1.PickQty + D2.PickQty > D4.Quantity
          ORDER BY D3.DocNum;


          IF @ItemCode IS NOT NULL
          BEGIN
              SELECT 1 AS ERROR, @ItemCode + ' already picked in other pick list(' + @PickDocNum + ').' AS ErrorMessage;
              RETURN;
          END
      END

      SELECT 0 AS ERROR, '' AS ErrorMessage;

  END
